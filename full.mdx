class Counter extends Component {
  constructor(props) {
    super(props);
    this.state = {
      count: 1,
    };
  }
  render() {
    const { count } = this.state;
    return (
      <div>
        <button
          onClick={(_) => this.setState((c) => c - 1)}
        >
          Less
        </button>
        <h1>{count}</h1>
        <button
          onClick={(_) => this.setState((c) => c + 1)}
        >
          More
        </button>
      </div>
    );
  }
}
